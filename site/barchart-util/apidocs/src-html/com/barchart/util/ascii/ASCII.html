164            public static final String toHexString(byte[] array) {
165                    final int size = array.length;
166                    final byte[] hexArray = new byte[size * 2];
167                    int i = 0;
168                    for (int k = 0; k < size; k++) {
169                            byte value = array[k];
170                            byte hex0 = (byte) ((value & 0xF0) >>> 4);
171                            byte hex1 = (byte) ((value & 0x0F));
172                            hexArray[i++] = hexValues[hex0].code;
173                            hexArray[i++] = hexValues[hex1].code;
174                    }
175                    return new String(hexArray, ASCII_CHARSET);
176            }

178            public static final void toUpperCase(final byte[] array) {
179                    if (array == null) {
180                            return;
181                    }
182                    final int size = array.length;
183                    for (int k = 0; k < size; k++) {
184                            final byte ascii = array[k];
185                            if (_a_ <= ascii && ascii <= _z_) {
186                                    array[k] = (byte) (ascii - 32);
187                            }
188                    }
189            }
190    
191            public static final void toLowerCase(final byte[] array) {
192                    if (array == null) {
193                            return;
194                    }
195                    final int size = array.length;
196                    for (int k = 0; k < size; k++) {
197                            final byte ascii = array[k];
198                            if (_A_ <= ascii && ascii <= _Z_) {
199                                    array[k] = (byte) (ascii + 32);
200                            }
201                    }
202            }

260            public static final boolean containsDigit(final byte[] array) {
261                    if (array == null) {
262                            return false;
263                    }
264                    final int size = array.length;
265                    for (int k = 0; k < size; k++) {
266                            if (isDigit(array[k])) {
267                                    return true;
268                            }
269                    }
270                    return false;
271            }

299            public static final boolean containsDigit(final CharSequence sequence) {
300                    if (sequence == null) {
301                            return false;
302                    }
303                    final int size = sequence.length();
304                    for (int k = 0; k < size; k++) {
305                            if (isDigit(sequence.charAt(k))) {
