008    package com.barchart.util.thread;
009    
010    import java.io.InputStream;
011    import java.io.OutputStream;
012    import java.io.Reader;
013    import java.io.Writer;
014    import java.net.DatagramSocket;
015    import java.net.ServerSocket;
016    import java.net.Socket;
017    import java.util.LinkedList;
018    import java.util.List;
019    import java.util.concurrent.ThreadFactory;
020    import java.util.concurrent.atomic.AtomicInteger;
021    
022    import org.slf4j.Logger;
023    import org.slf4j.LoggerFactory;
024    
025    import com.barchart.util.anno.ThreadSafe;
026    
027    /**
028     * use this thread implementation to enable InterruptedException - like behavior
029     * for IO methods that block on input/output but ignore Thread.interrupt() call;
030     * such as DatagramSocket.receive(), InputStream.read(), etc.; thread target
031     * must be of a supported type : {@link java.net.ServerSocket},
032     * {@link java.net.Socket}, {@link java.net.DatagramSocket},
033     * {@link java.io.InputStream}, {@link java.io.OutputStream},
034     * {@link java.io.Reader}, {@link java.io.Writer};
035     */
036    @ThreadSafe
037    public class ThreadInterruptibleIO extends Thread {
038    
039            /** target wrapper */
040            private interface Interrupter {
041                    /** one time action called on ThreadInterruptibleIO.interrupt */
042                    void fire();
043            }
044    
045            private final static Logger log = LoggerFactory
046                            .getLogger(ThreadInterruptibleIO.class);
047    
048            /** guarded by "this"; one list per thread */
049            private final List<Interrupter> interrupterList = new LinkedList<Interrupter>();
050    
051            public ThreadInterruptibleIO(final Runnable task) {
052                    super(task);
053            }
054    
055            /** target must be of supported type */
056            private final void registerTarget(final Object target) {
057    
058                    final Interrupter interrupter;
059    
060                    if (target == null) {
061                            log.error("null target; ignoring");
062                            return;
063                    } else if (target instanceof ServerSocket) {
064                            interrupter = new Interrupter() {
065    
066                                    final ServerSocket socket = (ServerSocket) target;
067    
068                                    @Override
069                                    public void fire() {
070                                            try {
071                                                    socket.close();
072                                            } catch (final Exception e) {
073                                                    log.error("unexpected; ignoring", e);
074                                            }
075                                    }
076    
077                                    @Override
078                                    public String toString() {
079                                            return socket.toString();
080                                    }
081                            };
082                    } else if (target instanceof Socket) {
083                            interrupter = new Interrupter() {
084    
085                                    final Socket socket = (Socket) target;
086    
087                                    @Override
088                                    public void fire() {
089                                            try {
090                                                    socket.close();
091                                            } catch (final Exception e) {
092                                                    log.error("unexpected; ignoring", e);
093                                            }
094                                    }
095    
096                                    @Override
097                                    public String toString() {
098                                            return socket.toString();
099                                    }
100                            };
101                    } else if (target instanceof DatagramSocket) {
102                            interrupter = new Interrupter() {
103    
104                                    final DatagramSocket socket = (DatagramSocket) target;
105    
106                                    @Override
107                                    public void fire() {
108                                            socket.close();
109                                    }
110    
111                                    @Override
112                                    public String toString() {
113                                            return "DatagramSocket[ local=" + socket.getLocalAddress()
114                                                            + " remote=" + socket.getInetAddress() + " ]";
115                                    }
116                            };
117                    } else if (target instanceof InputStream) {
118                            interrupter = new Interrupter() {
119    
120                                    final InputStream stream = (InputStream) target;
121    
122                                    @Override
123                                    public void fire() {
124                                            try {
125                                                    stream.close();
126                                            } catch (final Exception e) {
127                                                    log.error("unexpected; ignoring", e);
128                                            }
129                                    }
130    
131                                    @Override
132                                    public String toString() {
133                                            return stream.toString();
134                                    }
135                            };
136                    } else if (target instanceof OutputStream) {
137                            interrupter = new Interrupter() {
138    
139                                    final OutputStream stream = (OutputStream) target;
140    
141                                    @Override
142                                    public void fire() {
143                                            try {
144                                                    stream.close();
145                                            } catch (final Exception e) {
146                                                    log.error("unexpected; ignoring", e);
147                                            }
148                                    }
149    
150                                    @Override
151                                    public String toString() {
152                                            return stream.toString();
153                                    }
154                            };
155                    } else if (target instanceof Reader) {
156                            interrupter = new Interrupter() {
157    
158                                    final Reader stream = (Reader) target;
159    
160                                    @Override
161                                    public void fire() {
162                                            try {
163                                                    stream.close();
164                                            } catch (final Exception e) {
165                                                    log.error("unexpected; ignoring", e);
166                                            }
167                                    }
168    
169                                    @Override
170                                    public String toString() {
171                                            return stream.toString();
172                                    }
173                            };
174                    } else if (target instanceof Writer) {
175                            interrupter = new Interrupter() {
176    
177                                    final Writer stream = (Writer) target;
178    
179                                    @Override
180                                    public void fire() {
181                                            try {
182                                                    stream.close();
183                                            } catch (final Exception e) {
184                                                    log.error("unexpected; ignoring", e);
185                                            }
186                                    }
187    
188                                    @Override
189                                    public String toString() {
190                                            return stream.toString();
191                                    }
192                            };
193                    } else {
194    
195                            log.error("unsupported target type; ignoring; class : {}", target
196                                            .getClass().getName());
197                            return;
198    
199                    }
200    
201                    synchronized (interrupterList) {
202                            if (interrupterList.contains(target)) {
203                                    log.error("trying to add duplicate target; ignoring");
204                                    return;
205                            }
206                            interrupterList.add(interrupter);
207                    }
208    
209                    log.debug("added interrupter : {}", interrupter);
210    
211            }
212    
213            /**
214             * will clear interrupted targets from registration list after the interrupt
215             * event
216             */
217            @Override
218            public final void interrupt() {
219    
220                    // propagate interrupt
221                    super.interrupt();
222    
223                    synchronized (interrupterList) {
224                            for (final Interrupter interrupter : interrupterList) {
225                                    interrupter.fire();
226                                    log.debug("fired interrupter : {}", interrupter);
227                            }
228                            interrupterList.clear();
229                    }
230    
231            }
232    
233            /**
234             * target must be of supported type
235             */
236            public final void addTarget(final Object target) {
237                    registerTarget(target);
238            }
239    
240            /**
241             * note that interrupt event will auto clear all thread targets; call this
242             * if you want to disable target before the interrupt occurred
243             */
244            public final void removeTarget(final Object target) {
245                    unregisterTarget(target);
246            }
247    
248            /**
249             * must be called from ThreadInterruptibleIO thread; target must be of
250             * supported type;
251             */
252            public static final void addTargetToCurrentThread(final Object target) {
253    
254                    final Thread currentThread = Thread.currentThread();
255    
256                    if (currentThread instanceof ThreadInterruptibleIO) {
257    
258                            final ThreadInterruptibleIO interruptableThread = //
259                            (ThreadInterruptibleIO) currentThread;
260    
261                            interruptableThread.addTarget(target);
262    
263                    } else {
264                            log.error("unsupported operation; ignoring; thread class : {}",
265                                            currentThread.getClass().getName());
266                    }
267    
268            }
269    
270            private static final AtomicInteger factoryThreadCount = new AtomicInteger(0);
271    
272            private static final ThreadFactory factory = new ThreadFactory() {
273                    @Override
274                    public Thread newThread(final Runnable task) {
275                            final Thread thread = new ThreadInterruptibleIO(task);
276                            thread.setName("InterruptibleThread-"
277                                            + factoryThreadCount.getAndIncrement());
278                            return thread;
279                    }
280            };
281    
282            /**
283             * executors must use this factory, so that tasks can add their
284             * interruptible IO targets;
285             */
286            public static final ThreadFactory getFactory() {
287                    return factory;
288            }
289    
290            /**
291             * must be called from ThreadInterruptibleIO thread
292             */
293            public static final void removeTargetFromCurrentThread(final Object target) {
294    
295                    if (target == null) {
296                            log.error("null target; ignoring");
297                            return;
298                    }
299    
300                    final Thread currentThread = Thread.currentThread();
301    
302                    if (currentThread instanceof ThreadInterruptibleIO) {
303    
304                            final ThreadInterruptibleIO interruptibleThread = //
305                            (ThreadInterruptibleIO) currentThread;
306    
307                            interruptibleThread.removeTarget(target);
308    
309                    } else {
310                            log.error("unsupported operation; ignoring; thread class : {}",
311                                            currentThread.getClass().getName());
312                    }
313    
314            }
315    
316            private final void unregisterTarget(final Object target) {
317    
318                    if (target == null) {
319                            log.error("null target; ignoring");
320                            return;
321                    }
322    
323                    synchronized (interrupterList) {
324                            while (interrupterList.remove(target)) {
325                                    // cleanup
326                            }
327                    }
328    
329            }
330    
331    }
