001    package com.barchart.util.test;
002    
003    import java.lang.reflect.Field;
004    import java.util.Collection;
005    import java.util.concurrent.Callable;
006    import java.util.concurrent.TimeoutException;
007    
008    public final class CallableTest {
009    
010            private CallableTest() {
011            }
012    
013            /**
014             * Defaults to wait for ten seconds.
015             */
016            public static void waitFor(final Callable<Boolean> condition)
017                            throws Exception {
018                    waitFor(condition, null, 1000);
019            }
020    
021            /**
022             * Defaults to wait for ten seconds.
023             */
024            public static void waitFor(final Callable<Boolean> condition,
025                            final String message) throws Exception {
026                    waitFor(condition, message, 1000);
027            }
028    
029            /**
030             * Defaults to wait for ten seconds.
031             */
032            public static void waitFor(final Callable<Boolean> condition,
033                            final long maxTime) throws Exception {
034                    waitFor(condition, null, maxTime);
035            }
036    
037            public static void waitFor(final Callable<Boolean> condition,
038                            final String message, final long maxTime) throws Exception {
039                    waitFor(condition, message, maxTime, 1);
040            }
041    
042            public static void waitFor(final Callable<Boolean> condition,
043                            String message, final long maxTime, final long pollInterval)
044                            throws Exception {
045    
046                    if (message == null) {
047                            message = "waitFor() time expired before assertion passed";
048                    }
049    
050                    final long start = System.currentTimeMillis();
051    
052                    while (!condition.call()) {
053                            if (System.currentTimeMillis() > start + maxTime) {
054                                    throw new TimeoutException(message);
055                            }
056                            Thread.sleep(pollInterval);
057                    }
058    
059            }
060    
061            public static class CollectionSize implements Callable<Boolean> {
062    
063                    private final Collection<?> collection;
064                    private final int expectedSize;
065    
066                    public CollectionSize(final Collection<?> collection_,
067                                    final int expectedSize_) {
068                            collection = collection_;
069                            expectedSize = expectedSize_;
070                    }
071    
072                    @Override
073                    public Boolean call() throws Exception {
074                            return collection.size() == expectedSize;
075                    }
076    
077            }
078    
079            public static class FieldValue implements Callable<Boolean> {
080    
081                    private final Object obj;
082                    private final Object value;
083    
084                    private final Field field;
085    
086                    public FieldValue(final Object obj_, final String fieldName_,
087                                    final Object value_) throws NoSuchFieldException,
088                                    SecurityException {
089    
090                            obj = obj_;
091                            value = value_;
092    
093                            field = obj.getClass().getField(fieldName_);
094    
095                    }
096    
097                    @Override
098                    public Boolean call() throws Exception {
099                            if (value != null) {
100                                    return value.equals(field.get(obj));
101                            }
102                            return value == field.get(obj);
103                    }
104    
105            }
106    
107    }
