008    package com.barchart.util.enums;
009    
010    public abstract class ParamEnumBase2D<V, T extends ParamEnum2D<V, T>> extends
011                    ParamEnumBase<V, T> implements ParamEnum2D<V, T> {
012    
013            //
014    
015            protected final int row;
016    
017            @Override
018            public final int row() {
019                    return row;
020            }
021    
022            protected final int col;
023    
024            @Override
025            public final int col() {
026                    return col;
027            }
028    
029            protected ParamEnumBase2D(final ParamEnumBase<?, ?>[] values,
030                            final int ordinal, final V defVal, final int row, final int col) {
031                    super(values, ordinal, defVal);
032                    this.row = row;
033                    this.col = col;
034            }
035    
036    }
