008    package com.barchart.util.enums;
009    
010    /** type safe ENUM key */
011    public interface Dict<V> extends EnumJDK {
012    
013            /**
014             * @return creation order; can change from one JVM run another; otherwise is
015             *         GUID during JVM run session;
016             */
017            int sequence();
018    
019            /**
020             * @return relative index inside class (and super class and *); immutable;
021             *         based on ENUM creation order inside inheritance hierarchy; based
022             *         on ENUM order inside each level;
023             * */
024            @Override
025            int ordinal();
026    
027            /**
028             * @return field name; unique inside each inheritance hierarchy
029             **/
030            @Override
031            String name();
032    
033            //
034    
035            /**
036             * @return immutable code for external serialization; user provided optional
037             *         parameter; not checked for uniqueness;
038             */
039            String code();
040    
041            /** @return unique key based on class names space */
042            String guid();
043    
044            /** equality is based on ordinal */
045            @Override
046            boolean equals(Object dict);
047    
048            /** hash code is based on ordinal */
049            @Override
050            int hashCode();
051    
052    }
