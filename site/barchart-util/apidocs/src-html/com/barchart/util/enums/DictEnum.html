013    import java.util.ArrayList;

018    import java.util.concurrent.ConcurrentMap;

021    /**
022     * base class for type safe dictionary enumerators
023     */
024    public class DictEnum<V> implements Dict<V>, Comment {
025    
026            {
027                    // System.out.println("LOAD DictEnum");
028            }
029    
030            private static final AtomicInteger SEQUENCE = new AtomicInteger(0);
031    
032            private static final ConcurrentMap<Integer, DictEnum<?>> ENUM_MAP = //
033            new ConcurrentHashMap<Integer, DictEnum<?>>();
034    
035            private static final ConcurrentMap<Class<DictEnum<?>>, AtomicInteger> ORDINAL_MAP = //
036            new ConcurrentHashMap<Class<DictEnum<?>>, AtomicInteger>();
037    
038            private static final ConcurrentMap<Class<DictEnum<?>>, DictEnum<?>[]> VALUES_MAP = //
039            new ConcurrentHashMap<Class<DictEnum<?>>, DictEnum<?>[]>();
040    
041            //
042    
043            /** @return clone array */
044            @SuppressWarnings("unchecked")
045            public static final <K extends DictEnum<?>> DictEnum<?>[] valuesFor(
046                            final Class<K> klaz) {
047    
048                    DictEnum<?>[] values = VALUES_MAP.get(klaz);
049    
050                    if (values == null) {
051    
052                            try {
053                                    /* initialize klaz, assign static fields */
054                                    final Constructor<?> ctor = klaz.getDeclaredConstructor();
055                                    ctor.setAccessible(true);
056                                    ctor.newInstance();
057                            } catch (Exception e) {
058                                    e.printStackTrace();
059                            }
060    
061                            final int size = ENUM_MAP.size();
062    
063                            final List<DictEnum<?>> target = new ArrayList<DictEnum<?>>(size);
064    
065                            final Set<String> nameSet = new HashSet<String>();
066    
067                            for (int k = 0; k < size; k++) {
068                                    final DictEnum<?> dict = ENUM_MAP.get(k);
069                                    if (dict.getClass().isAssignableFrom(klaz)) {
070                                            target.add(dict);
071                                            nameSet.add(dict.name());
072                                    }
073                            }
074    
075                            if (nameSet.size() != target.size()) {
076    
077                                    System.out.println("contains duplicate names : " + klaz);
078    
079                                    for (final DictEnum<?> dict : target) {
080                                            System.out.println(" name=" + dict.name() + " guid="
081                                                            + dict.guid());
082                                    }
083    
084                                    throw new RuntimeException("duplicate names in dict enum");
085    
086                            }
087    
088                            values = target.toArray(new DictEnum<?>[0]);
089    
090                            VALUES_MAP.put((Class<DictEnum<?>>) klaz, values);
091    
092                    }
093    
094                    return values.clone();
095    
096            }
097    
098            @SuppressWarnings("unchecked")
099            protected static final <K extends DictEnum<?>> K[] valuesForType(
100                            final Class<K> klaz) {
101    
102                    final DictEnum<?>[] source = valuesFor(klaz);
103    
104                    final int size = source.length;
105    
106                    final K[] target = (K[]) Array.newInstance(klaz, size);
107    
108                    for (int k = 0; k < size; k++) {
109                            target[k] = (K) source[k];
110                    }
111    
112                    return target;
113    
114            }

118            private String name;
119            private String guid;
120    
121            private final int sequence;
122            private final int ordinal;
123            private final String code;
124            private final String comment;
125    
126            //
127    
128            @SuppressWarnings("unchecked")
129            private int makeOrdinal() {
130    
131                    final Class<DictEnum<?>> klaz = (Class<DictEnum<?>>) getClass();
132    
133                    final Class<?> parent = klaz.getSuperclass();
134    
135                    // System.out.println("parent=" + parent);
136    
137                    AtomicInteger parentCount = ORDINAL_MAP.get(parent);
138                    if (parentCount == null) {
139                            parentCount = new AtomicInteger(0);
140                            ORDINAL_MAP.put((Class<DictEnum<?>>) parent, parentCount);
141                    }
142    
143                    // System.out.println("parentCount=" + parentCount);
144    
145                    final AtomicInteger countOld = ORDINAL_MAP.get(klaz);
146    
147                    if (countOld == null) {
148    
149                            // initial instance
150    
151                            final int indexStart;
152    
153                            if (parent == DictEnum.class) {
154                                    indexStart = 0;
155                            } else {
156                                    indexStart = parentCount.get();
157                            }
158    
159                            final AtomicInteger countNew = new AtomicInteger(indexStart);
160    
161                            ORDINAL_MAP.put(klaz, countNew);
162    
163                            return countNew.getAndIncrement();
164    
165                    } else {
166    
167                            // following instance
168    
169                            return countOld.getAndIncrement();
170    
171                    }
172            }
173    
174            private int makeSequence() {
175                    return SEQUENCE.getAndIncrement();
176            }
177    
178            /** use for class load only */
179            protected DictEnum() {
180                    this.code = null;
181                    this.comment = null;
182                    this.sequence = -1;
183                    this.ordinal = -1;
184            }
185    
186            protected DictEnum(final String comment) {
187                    this("", comment);
188            }
189    
190            protected DictEnum(final String code, final String comment) {
191                    this.code = code;
192                    this.comment = comment;
193                    this.sequence = makeSequence();
194                    this.ordinal = makeOrdinal();
195                    ENUM_MAP.put(sequence, this);
196            }
197    
198            @Override
199            public final String name() {
200                    if (name == null) {
201                            final Field[] fields = getClass().getDeclaredFields();
202                            for (final Field field : fields) {
203                                    try {
204                                            field.setAccessible(true);
205                                            if (this == field.get(this)) {
206                                                    name = field.getName();
207                                                    break;
208                                            }
209                                    } catch (final Exception e) {
210                                            e.printStackTrace();
211                                    }
212                            }
213                            if (name == null) {
214                                    name = "" + ordinal();
215                            }
216                    }
217                    return name;
218            }
219    
220            @Override
221            public final String guid() {
222                    if (guid == null) {
223                            guid = getClass().getName() + "#" + name();
224                    }
225                    return guid;
226            }
227    
228            @Override
229            public final int sequence() {
230                    return sequence;
231            }
232    
233            @Override
234            public final int ordinal() {
235                    return ordinal;
236            }
237    
238            @Override
239            public final boolean equals(final Object dict) {
240                    if (dict instanceof DictEnum) {
241                            final DictEnum<?> that = (DictEnum<?>) dict;
242                            return that.sequence() == this.sequence();
243                    }
244                    return false;
245            }
246    
247            @Override
248            public final int hashCode() {
249                    return sequence();
250            }
251    
252            @Override
253            public String code() {
254                    return code;
255            }
256    
257            @Override
258            public final String getComment() {
259                    return comment;
260            }
261    
262            @Override
263            public final void setComment(String comment) {
264                    new UnsupportedOperationException("do not use").printStackTrace();
265            }
266    
267            private String text;
268    
269            @Override
270            public String toString() {
271                    if (text == null) {
272                            text = String
273                                            .format("NAME=%-20s ORDINAL=%02d SEQUENCE=%02d GUID=%30s  COMMENT=%s ",
274                                                            name(), ordinal(), sequence(), guid(), getComment());
275                    }
276                    return text;
277            }
