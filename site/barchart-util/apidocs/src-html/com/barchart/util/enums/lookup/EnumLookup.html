008    package com.barchart.util.enums.lookup;
009    
010    import java.util.HashMap;
011    import java.util.Map;
012    
013    public final class EnumLookup<K, V extends Enum<?> & Keyable<K>> {
014    
015            private final Map<K, V> map;
016    
017            private EnumLookup(Map<K, V> map) {
018                    this.map = map;
019            }
020    
021            public V get(K key) {
022                    return map.get(key);
023            }
024    
025            public static <K, V extends Enum<?> & Keyable<K>> EnumLookup<K, V> create(Class<V> enumClass) {
026                    V[] enumConstants = enumClass.getEnumConstants();
027                    Map<K, V> map = new HashMap<K, V>();
028                    for (V constant : enumConstants) {
029                            K key = constant.getKey();
030                            map.put(key, constant);
031                    }
032                    return new EnumLookup<K, V>(map);
033            }
034    
035    }
