001    package com.barchart.util.enums.lookup;
002    
003    import java.util.HashMap;
004    import java.util.Map;
005    
006    public final class EnumLookup<K, V extends Enum<?> & Keyable<K>> {
007    
008            private final Map<K, V> map;
009    
010            private EnumLookup(Map<K, V> map) {
011                    this.map = map;
012            }
013    
014            public V get(K key) {
015                    return map.get(key);
016            }
017    
018            public static <K, V extends Enum<?> & Keyable<K>> EnumLookup<K, V> create(Class<V> enumClass) {
019                    V[] enumConstants = enumClass.getEnumConstants();
020                    Map<K, V> map = new HashMap<K, V>();
021                    for (V constant : enumConstants) {
022                            K key = constant.getKey();
023                            map.put(key, constant);
024                    }
025                    return new EnumLookup<K, V>(map);
026            }
027    
028    }
