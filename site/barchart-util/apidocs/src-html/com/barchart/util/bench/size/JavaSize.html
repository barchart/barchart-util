008    package com.barchart.util.bench.size;
009    
010    import java.lang.reflect.Array;
011    import java.lang.reflect.Field;
012    import java.lang.reflect.Modifier;
013    import java.util.Deque;
014    import java.util.concurrent.LinkedBlockingDeque;
015    
016    /**
017     * FIXME 32 bit only
018     * 
019     * TODO support both 32 and 64 and OOPS
020     * 
021     * http://wikis.sun.com/display/HotSpotInternals/CompressedOops
022     * 
023     */
024    public final class JavaSize {
025    
026            private JavaSize() {
027            }
028    
029            // 32 bit java only
030            public static final int REFERENCE_SIZE = 4;
031    
032            // 32 bit java only
033            public static final int OBJECT_SIZE = 8;
034    
035            // 32 bit java only
036            public static final int OBJECT_WORD = 8;
037    
038            public static final int of(final boolean b) {
039                    return 1;
040            }
041    
042            public static final int of(final byte b) {
043                    return 1;
044            }
045    
046            public static final int of(final char c) {
047                    return 2;
048            }
049    
050            public static final int of(final short s) {
051                    return 2;
052            }
053    
054            public static final int of(final int i) {
055                    return 4;
056            }
057    
058            public static final int of(final long l) {
059                    return 8;
060            }
061    
062            public static final int of(final float f) {
063                    return 4;
064            }
065    
066            public static final int of(final double d) {
067                    return 8;
068            }
069    
070            private static final int sizeNonArray(final Class<?> klaz,
071                            final Object object) {
072    
073                    final Deque<Object> stack = STACK.get();
074    
075                    stack.push(object);
076    
077                    int size = 0;
078    
079                    final Class<?> superKlaz = klaz.getSuperclass();
080    
081                    if (superKlaz == null) {
082                            size += OBJECT_SIZE;
083                            // log.debug("root1={} size={}", "Object", size);
084                    } else {
085                            size += sizeNonArray(superKlaz, object);
086                            // log.debug("root2={} size={}", superKlaz.getName(), size);
087                    }
088    
089                    int objFieldsSize = 0;
090    
091                    // no inherited
092                    final Field fieldArray[] = klaz.getDeclaredFields();
093    
094                    for (final Field field : fieldArray) {
095    
096                            final int modifiers = field.getModifiers();
097    
098                            if ((modifiers & Modifier.STATIC) != 0) {
099                                    continue;
100                            }
101    
102                            final Class<?> type = field.getType();
103    
104                            final int fieldSize = sizePrimitive(type);
105    
106                            // will count references
107                            objFieldsSize += fieldSize;
108                            // log.debug("field={} size={}", field.getName(), fieldSize);
109    
110                            if (type.isPrimitive()) {
111                                    continue;
112                            }
113    
114                            field.setAccessible(true);
115    
116                            Object value = null;
117                            try {
118                                    value = field.get(object);
119                            } catch (final Exception e) {
120                                    e.printStackTrace();
121                            }
122    
123                            if (value == null) {
124                                    continue;
125                            }
126    
127                            if (value == object) {
128                                    // ignore self references
129                                    continue;
130                            }
131    
132                            if (stack.contains(value)) {
133                                    // ignore self references
134                                    continue;
135                            }
136    
137                            if (type.isArray()) {
138                                    size += sizeArray(type, value);
139                            } else {
140                                    size += sizeNonArray(type, value);
141                            }
142    
143                    }
144    
145                    // reflect word alignment
146                    final int wordPart = objFieldsSize % OBJECT_WORD;
147                    if (wordPart != 0) {
148                            objFieldsSize -= wordPart;
149                            objFieldsSize += OBJECT_WORD;
150                    }
151    
152                    size += objFieldsSize;
153    
154                    stack.pop();
155    
156                    return size;
157    
158            }
159    
160            private static final int sizePrimitive(final Class<?> klaz) {
161                    if (klaz == Boolean.TYPE)
162                            return 1;
163                    else if (klaz == Byte.TYPE)
164                            return 1;
165                    else if (klaz == Character.TYPE)
166                            return 2;
167                    else if (klaz == Short.TYPE)
168                            return 2;
169                    else if (klaz == Integer.TYPE)
170                            return 4;
171                    else if (klaz == Long.TYPE)
172                            return 8;
173                    else if (klaz == Float.TYPE)
174                            return 4;
175                    else if (klaz == Double.TYPE)
176                            return 8;
177                    else if (klaz == Void.TYPE)
178                            return 0;
179                    else
180                            return REFERENCE_SIZE;
181            }
182    
183            private static final int sizeArray(final Class<?> klaz, final Object object) {
184    
185                    final Deque<Object> stack = STACK.get();
186    
187                    stack.push(object);
188    
189                    final Class<?> type = klaz.getComponentType();
190    
191                    final int length = Array.getLength(object);
192    
193                    int size = OBJECT_SIZE;
194    
195                    if (type.isPrimitive()) {
196    
197                            size += length * sizePrimitive(type);
198    
199                    } else {
200    
201                            for (int i = 0; i < length; i++) {
202    
203                                    size += REFERENCE_SIZE;
204    
205                                    final Object value = Array.get(object, i);
206    
207                                    if (value == null) {
208                                            continue;
209                                    }
210    
211                                    if (value == object) {
212                                            // ignore self references
213                                            continue;
214                                    }
215    
216                                    if (stack.contains(value)) {
217                                            // ignore self references
218                                            continue;
219                                    }
220    
221                                    final Class<?> itemKlaz = value.getClass();
222    
223                                    if (itemKlaz.isArray()) {
224                                            size += sizeArray(itemKlaz, value);
225                                    } else {
226                                            size += sizeNonArray(itemKlaz, value);
227                                    }
228    
229                            }
230    
231                    }
232    
233                    stack.pop();
234    
235                    return size;
236    
237            }
238    
239            private static final ThreadLocal<Deque<Object>> STACK = new ThreadLocal<Deque<Object>>() {
240                    @Override
241                    protected Deque<Object> initialValue() {
242                            return new LinkedBlockingDeque<Object>();
243                    }
244            };
245    
246            public static final int of(final Object object) {
247    
248                    if (object == null) {
249                            return 0;
250                    }
251    
252                    final Deque<Object> stack = STACK.get();
253    
254                    assert stack.isEmpty();
255    
256                    final Class<?> klaz = object.getClass();
257    
258                    final int size;
259    
260                    if (klaz.isArray()) {
261                            size = sizeArray(klaz, object);
262                    } else {
263                            size = sizeNonArray(klaz, object);
264                    }
265    
266                    assert stack.isEmpty();
267    
268                    return size;
269    
270            }
