008    package com.barchart.util.math;
009    
010    public final class MathExtra {
011    
012            private MathExtra() {
013    
014            }
015    
016            public static final byte castCharToByte(final char value)
017                            throws ArithmeticException {
018                    final byte result = (byte) value;
019                    if (result == value) {
020                            return result;
021                    } else {
022                            throw new ArithmeticException("can not cast char to byte");
023                    }
024            }
025    
026            public static final byte castIntToByte(final int value)
027                            throws ArithmeticException {
028                    final byte result = (byte) value;
029                    if (result == value) {
030                            return result;
031                    } else {
032                            throw new ArithmeticException("can not cast int to byte");
033                    }
034            }
035    
036            public static byte castLongToByte(final long value)
037                            throws ArithmeticException {
038                    final byte result = (byte) value;
039                    if (result == value) {
040                            return result;
041                    } else {
042                            throw new ArithmeticException("can not cast long to byte");
043                    }
044            }
045    
046            public static final int castLongToInt(final long value)
047                            throws ArithmeticException {
048                    final int result = (int) value;
049                    if (result == value) {
050                            return result;
051                    } else {
052                            throw new ArithmeticException("can not cast long to int");
053                    }
054            }
055    
056            public static final long longMult(final long a, final long b)
057                            throws ArithmeticException {
058                    final long r = a * b;
059                    if (a != 0 && r / a != b) {
060                            throw new ArithmeticException("overflow" + " a=" + a + " b=" + b
061                                            + " r=" + r);
062                    }
063                    return r;
064            }
065    
066            public static final long longMult10(final long v)
067                            throws ArithmeticException {
068                    final long r = v * 10L;
069                    if (r / 10L != v) {
070                            throw new ArithmeticException("overflow" + " v=" + v + " r=" + r);
071                    }
072                    return r;
073            }
074    
075            public static final int intAdd(final int a, final int b)
076                            throws ArithmeticException {
077                    final int r = a + b;
078                    if (((a ^ r) & (b ^ r)) < 0) {
079                            throw new ArithmeticException("overflow");
080                    }
081                    return r;
082            }
083    
084            public static final int intSub(final int a, final int b)
085                            throws ArithmeticException {
086                    final int r = a + (-b);
087                    if (((a ^ r) & ((-b) ^ r)) < 0) {
088                            throw new ArithmeticException("overflow");
089                    }
090                    return r;
091            }
092    
093            public static final long intMult(final int a, final int b)
094                            throws ArithmeticException {
095                    final int r = a * b;
096                    if (a != 0 && r / a != b) {
097                            throw new ArithmeticException("overflow" + " a=" + a + " b=" + b
098                                            + " r=" + r);
099                    }
100                    return r;
101            }
102    
103            public static final int intMult10(final int v) throws ArithmeticException {
104                    final int r = v * 10;
105                    if (r / 10 != v) {
106                            throw new ArithmeticException("overflow" + " v=" + v + " r=" + r);
107                    }
108                    return r;
109            }
110    
111            public static final long longAdd(final long a, final long b)
112                            throws ArithmeticException {
113                    final long r = a + b;
114                    if (((a ^ r) & (b ^ r)) < 0) {
115                            throw new ArithmeticException("overflow");
116                    }
117                    return r;
118            }
119    
120            public static final long longSub(final long a, final long b)
121                            throws ArithmeticException {
122                    final long r = a + (-b);
123                    if (((a ^ r) & ((-b) ^ r)) < 0) {
124                            throw new ArithmeticException("overflow");
125                    }
126                    return r;
127            }
128    
129    }
