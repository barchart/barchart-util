001    package com.barchart.util.concurrent;
002    
003    public class CancellableFutureNotifier<V, T extends FutureCallback<V, T>>
004                    extends FutureNotifierBase<V, T> {
005    
006            private FutureCallback<?, ?> parent;
007    
008            public CancellableFutureNotifier() {
009                    parent = null;
010            }
011    
012            public CancellableFutureNotifier(final FutureCallback<?, ?> parent_) {
013                    parent = parent_;
014            }
015    
016            public void setCancelCallback(final FutureCallback<?, ?> parent_) {
017                    parent = parent_;
018            }
019    
020            @Override
021            public boolean cancel(final boolean mayInterrupt) {
022                    if (parent != null) {
023                            parent.cancel(mayInterrupt);
024                    }
025                    return super.cancel(mayInterrupt);
026            }
027    
028    }
