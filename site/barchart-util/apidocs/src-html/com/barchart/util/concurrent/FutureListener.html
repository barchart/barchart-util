001    package com.barchart.util.concurrent;
002    
003    import java.util.concurrent.Future;
004    
005    /**
006     * <p>
007     * Defines an interface for {@link FutureCallbackTask} callback listeners.
008     * 
009     * @author jeremy
010     * @see FutureCallbackTask
011     * @param <E>
012     *            The type of object that the callback returns
013     */
014    public interface FutureListener<E> {
015    
016            /**
017             * Called when the deferred task completes successfully.
018             * 
019             * @param value
020             *            The return value from the deferred task
021             */
022            public void resultAvailable(Future<E> result) throws Exception;
023    
024    }
