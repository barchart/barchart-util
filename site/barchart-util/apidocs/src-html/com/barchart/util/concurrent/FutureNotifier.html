001    package com.barchart.util.concurrent;
002    
003    /**
004     * An implementation of FutureCallback that does not do any actual computation,
005     * but just serves as a communication proxy between the executor and listeners.
006     * 
007     * @author jeremy
008     * 
009     * @param <V>
010     *            The result type
011     */
012    public class FutureNotifier<V> extends FutureNotifierBase<V, FutureNotifier<V>> {
013    
014            public static <P> FutureNotifier<P> success(final P result) {
015                    return new FutureNotifier<P>().succeed(result);
016            }
017    
018            public static <P> FutureNotifier<P> failure(final Throwable error) {
019                    return new FutureNotifier<P>().fail(error);
020            }
021    
022    }
