008    package com.barchart.util.values.lang;
009    
010    import static java.lang.Math.max;
011    import static java.lang.Math.min;
012    
013    import com.barchart.util.anno.NotMutable;
014    import com.barchart.util.math.MathExtra;
015    import com.barchart.util.values.api.Value;
016    import com.barchart.util.values.provider.ValueFreezer;
017    
018    @NotMutable
019    public abstract class ScaledDecimalValue<T extends ScaledDecimal<T, F> & Value<T>, F extends ScaledDecimal<F, F> & Value<F>>
020                    extends ValueFreezer<T> implements ScaledDecimal<T, F> {
021    
022            protected abstract T result(long mantissa, int exponent);
023    
024            // yes, same type
025            @SuppressWarnings("unchecked")
026            @Override
027            public final T norm() {
028    
029                    long m = mantissa();
030                    int e = exponent();
031    
032                    final int p = e;
033    
034                    while (m % 10 == 0) {
035                            m /= 10;
036                            e++;
037                    }
038    
039                    if (p == e) {
040                            return (T) this;
041                    } else {
042                            return result(m, e);
043                    }
044    
045            }
046    
047            // yes, same type
048            @SuppressWarnings("unchecked")
049            @Override
050            public final T scale(final int exponent) throws ArithmeticException {
051    
052                    long m = mantissa();
053                    int e = exponent();
054    
055                    if (e == exponent) {
056                            return (T) this;
057                    }
058    
059                    while (e > exponent) {
060                            m = MathExtra.longMult10(m);
061                            e--;
062                    }
063                    while (e < exponent) {
064                            m /= 10;
065                            e++;
066                    }
067    
068                    return result(m, e);
069    
070            }
071    
072            @Override
073            public final int compareTo(final T that) {
074    
075                    int e1 = this.exponent();
076                    int e2 = that.exponent();
077    
078                    long m1 = this.mantissa();
079                    long m2 = that.mantissa();
080    
081                    if (e1 != e2) {
082                            if ((m1 ^ m2) < 0) {
083                                    // different sign
084                                    return m1 >= 0 ? +1 : -1;
085                            } else {
086                                    // identical sign
087                                    final int e = max(e1, e2);
088                                    while (e > e1) {
089                                            m1 /= 10L;
090                                            e1++;
091                                    }
092                                    while (e > e2) {
093                                            m2 /= 10L;
094                                            e2++;
095                                    }
096                            }
097    
098                    }
099    
100                    return m1 < m2 ? -1 : (m1 == m2 ? 0 : +1);
101    
102            }
103    
104            @Override
105            public final long count(final T that) throws ArithmeticException {
106    
107                    int e1 = this.exponent();
108                    int e2 = that.exponent();
109    
110                    long m1 = this.mantissa();
111                    long m2 = that.mantissa();
112    
113                    if (e1 != e2) {
114                            final int e = Math.min(e1, e2);
115                            while (e < e1) {
116                                    m1 = MathExtra.longMult10(m1);
117                                    e1--;
118                            }
119                            while (e < e2) {
120                                    m2 = MathExtra.longMult10(m2);
121                                    e2--;
122                            }
123                    }
124    
125                    // return safeCastToInt(m1 / m2);
126                    return (m1 / m2);
127    
128            }
129    
130            @Override
131            public final int hashCode() {
132                    final long m = mantissa();
133                    final int e = exponent();
134                    return e ^ (int) (m ^ (m >>> 32));
135            }
136    
137            @Override
138            public String toString() {
139                    int e = exponent();
140                    double d  = mantissa() * Math.pow(10.0, e);
141                    
142                    int numberOfDecimalPlaces = e < 0 ? - e : 0;
143                    switch (numberOfDecimalPlaces) {
144                            case 0:
145                                    return String.format("%.0f", d);
146                            case 1:
147                                    return String.format("%.1f", d);
148                            case 2:
149                                    return String.format("%.2f", d);
150                            case 3:
151                                    return String.format("%.3f", d);
152                            case 4:
153                                    return String.format("%.4f", d);
154                            case 5:
155                                    return String.format("%.5f", d);
156                            default:
157                                    return String.format("%.6f", d);
158                    }
159                    
160            }
161    
162            // static final long SIGN = 1L << Long.SIZE;
163    
164            @Override
165            public final T add(final T that) throws ArithmeticException {
166    
167                    int e1 = this.exponent();
168                    int e2 = that.exponent();
169    
170                    long m1 = this.mantissa();
171                    long m2 = that.mantissa();
172    
173                    if (e1 != e2) {
174                            final int e = min(e1, e2);
175                            while (e < e1) {
176                                    m1 = MathExtra.longMult10(m1);
177                                    e1--;
178                            }
179                            while (e < e2) {
180                                    m2 = MathExtra.longMult10(m2);
181                                    e2--;
182                            }
183                    }
184    
185                    return result(MathExtra.longAdd(m1, m2), e1);
186    
187            }
188    
189            @Override
190            public final T sub(final T that) throws ArithmeticException {
191    
192                    int e1 = this.exponent();
193                    int e2 = that.exponent();
194    
195                    long m1 = this.mantissa();
196                    long m2 = that.mantissa();
197    
198                    if (e1 != e2) {
199                            final int e = min(e1, e2);
200                            while (e < e1) {
201                                    m1 = MathExtra.longMult10(m1);
202                                    e1--;
203                            }
204                            while (e < e2) {
205                                    m2 = MathExtra.longMult10(m2);
206                                    e2--;
207                            }
208                    }
209    
210                    return result(MathExtra.longSub(m1, m2), e1);
211    
212            }
213    
214            @Override
215            public final T mult(final long factor) throws ArithmeticException {
216    
217                    return result(MathExtra.longMult(mantissa(), factor), exponent());
218    
219            }
220    
221            @Override
222            public final T div(final long factor) throws ArithmeticException {
223    
224                    long m = this.mantissa();
225                    int e = this.exponent();
226    
227                    while (true) {
228                            final long r = m * 10L;
229                            if (r / 10L != m) {
230                                    break;
231                            }
232                            m = r;
233                            e--;
234                    }
235    
236                    return result(m / factor, e);
237    
238            }
239    
240            @Override
241            public final T neg() {
242    
243                    return result(-mantissa(), exponent());
244    
245            }
246    
247            @Override
248            public final boolean equalsScale(final T that) {
249                    return this.exponent() == that.exponent();
250            }
251    
252            @Override
253            public final boolean isZero() {
254                    return this.mantissa() == 0L;
255            }
256    
257            @Override
258            public final T mult(final F that) throws ArithmeticException {
259    
260                    final int e1 = this.exponent();
261                    final int e2 = that.exponent();
262    
263                    final long m1 = this.mantissa();
264                    final long m2 = that.mantissa();
265    
266                    return result(MathExtra.longMult(m1, m2), e1 + e2);
267    
268            }
269    
270            @Override
271            public final T div(final F that) throws ArithmeticException {
272    
273                    int e1 = this.exponent();
274                    final int e2 = that.exponent();
275    
276                    long m1 = this.mantissa();
277                    final long m2 = that.mantissa();
278    
279                    while (true) {
280                            final long r = m1 * 10L;
281                            if (r / 10L != m1) {
282                                    break;
283                            }
284                            m1 = r;
285                            e1--;
286                    }
287    
288                    return result(m1 / m2, e1 - e2);
289    
290            }
291    
292            @Override
293            public boolean equals(final Object thatValue) {
294                    throw new UnsupportedOperationException(
295                                    "must override in sub type in order to narrow closure");
296            }
297    
298    }
