008    package com.barchart.util.values.json;
009    
010    import org.codehaus.jackson.Version;
011    import org.codehaus.jackson.map.Module;
012    import org.codehaus.jackson.map.module.SimpleModule;
013    
014    import com.barchart.util.values.api.DecimalValue;
015    import com.barchart.util.values.api.PriceValue;
016    import com.barchart.util.values.api.SizeValue;
017    import com.barchart.util.values.api.TextValue;
018    import com.barchart.util.values.api.TimeValue;
019    
020    public class ValueModule {
021    
022            public static final String COLON = ":";
023            public static final String EXP = "E";
024            public static final String AT = "@";
025    
026            public static Module build() {
027    
028                    Version version = new Version(1, 0, 0, null);
029    
030                    SimpleModule module = new SimpleModule("values", version);
031    
032                    //
033    
034                    Class<DecimalValue> klazDecimal = DecimalValue.class;
035                    module.addSerializer(new DecimalValueSer(klazDecimal));
036                    module.addDeserializer(klazDecimal, new DecimalValueDes(klazDecimal));
037    
038                    //
039    
040                    Class<PriceValue> klazPrice = PriceValue.class;
041                    module.addSerializer(new PriceValueSer(klazPrice));
042                    module.addDeserializer(klazPrice, new PriceValueDes(klazPrice));
043    
044                    //
045    
046                    Class<SizeValue> klazSize = SizeValue.class;
047                    module.addSerializer(new SizeValueSer(klazSize));
048                    module.addDeserializer(klazSize, new SizeValueDes(klazSize));
049    
050                    //
051    
052                    Class<TextValue> klazText = TextValue.class;
053                    module.addSerializer(new TextValueSer(klazText));
054                    module.addDeserializer(klazText, new TextValueDes(klazText));
055    
056                    //
057    
058                    Class<TimeValue> klazTime = TimeValue.class;
059                    module.addSerializer(new TimeValueSer(klazTime));
060                    module.addDeserializer(klazTime, new TimeValueDes(klazTime));
061    
062                    //
063    
064                    return module;
065    
066            }
067    
068    }
