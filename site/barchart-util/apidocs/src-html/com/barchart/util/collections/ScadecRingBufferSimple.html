008    package com.barchart.util.collections;
009    
010    import com.barchart.util.values.lang.ScaledDecimal;
011    
012    public class ScadecRingBufferSimple<K extends ScaledDecimal<K, ?>, V> extends
013                    ScadecRingBufferBase<K, V> {
014    
015            @SuppressWarnings("unchecked")
016            public ScadecRingBufferSimple(final int size, final K keyStep) {
017                    checkKeyStep(keyStep);
018                    this.keyStep = keyStep;
019                    if (size <= 0) {
020                            throw new IllegalArgumentException("size <= 0");
021                    }
022                    this.array = (V[]) new Object[size];
023            }
024    
025            protected final V[] array;
026    
027            protected final K keyStep;
028    
029            @Override
030            public K keyStep() {
031                    return keyStep;
032            }
033    
034            @Override
035            public final int length() {
036                    return array.length;
037            }
038    
039            @Override
040            protected final V arrayGet(final int clue) {
041                    return array[clue];
042            }
043    
044            @Override
045            protected final void arraySet(final int clue, final V value) {
046                    array[clue] = value;
047            }
048    
049            @Override
050            protected final boolean isEmpty(final int clue) {
051                    return array[clue] == null;
052            }
053    
054    }
