008    package com.barchart.util.collections;
009    
010    // enum represented as long bit set
011    public interface BitSetEnum<E> {
012    
013            long NUL = 0L;
014    
015            long ONE = 1L;
016    
017            // max enum size
018            int LIMIT = 64;
019    
020            //
021    
022            /** ENUM create order */
023            int ordinal();
024    
025            //
026    
027            /**
028             * implements must provide:
029             * 
030             * private final long mask = ONE << index();
031             * 
032             * public final long mask() { return mask; }
033             * 
034             **/
035            long mask();
036    
037    }
