008    package com.barchart.util.collections;
009    
010    import java.util.ArrayList;
011    import java.util.Collection;
012    import java.util.List;
013    import java.util.Map;
014    import java.util.Map.Entry;
015    import java.util.Set;
016    
017    import com.barchart.util.anno.NotThreadSafe;
018    import com.barchart.util.anno.ThreadSafe;
019    import com.barchart.util.thread.Runner;
020    import com.barchart.util.thread.RunnerLoop;
021    
022    @NotThreadSafe
023    public class FastEnumMap<K extends BitSetEnum<K>, V> extends FastEnumBase<K>
024                    implements RunnerLoop<Entry<K, V>>, Map<K, V> {
025    
026            protected final FastEnumSet<K> enumSet = new FastEnumSet<K>(enumValues);
027    
028            @SuppressWarnings("unchecked")
029            protected final V[] values = (V[]) new Object[enumValues.length];
030    
031            protected FastEnumMap(final K[] values) {
032                    super(values);
033            }
034    
035            //
036    
037            public final long bitSet() {
038                    return enumSet.bitSet();
039            }
040    
041            @ThreadSafe(rule = "does not reflect concurrent modifications")
042            @Override
043            public <R> void runLoop(final Runner<R, Entry<K, V>> mapTask,
044                            final List<R> list) {
045    
046                    final MapEntry<K, V> entry = new MapEntry<K, V>();
047    
048                    final Runner<R, K> setTask = new Runner<R, K>() {
049                            @Override
050                            public R run(final K key) {
051                                    final V value = values[key.ordinal()];
052                                    if (value == null) {
053                                            // due to concurrent modifications
054                                            return null;
055                                    } else {
056                                            entry.key = key;
057                                            entry.value = value;
058                                            return mapTask.run(entry);
059                                    }
060                            }
061                    };
062    
063                    enumSet.runLoop(setTask, list);
064    
065            }
066    
067            @Override
068            public final boolean isEmpty() {
069                    return enumSet.isEmpty();
070            }
071    
072            @Override
073            public final V get(final Object key) {
074                    return values[((BitSetEnum<?>) key).ordinal()];
075            }
076    
077            @Override
078            public final V put(final K key, final V newValue) {
079                    final int ordinal = key.ordinal();
080                    final V oldValue = values[ordinal];
081                    values[ordinal] = newValue;
082                    enumSet.add(key);
083                    return oldValue;
084    
085            }
086    
087            @Override
088            public final V remove(final Object key) {
089                    final int ordinal = ((BitSetEnum<?>) key).ordinal();
090                    final V oldValue = values[ordinal];
091                    values[ordinal] = null;
092                    enumSet.remove(key);
093                    return oldValue;
094            }
095    
096            protected final Runner<Void, K> taskClear = //
097            new Runner<Void, K>() {
098                    @Override
099                    public Void run(final K key) {
100                            values[key.ordinal()] = null;
101                            return null;
102                    }
103            };
104    
105            @Override
106            public final void clear() {
107                    enumSet.runLoop(taskClear, null);
108                    enumSet.clear();
109            }
110    
111            @Override
112            public int size() {
113                    return enumSet.size();
114            }
115    
116            protected final Runner<V, K> taskGet = //
117            new Runner<V, K>() {
118                    @Override
119                    public V run(final K key) {
120                            return values[key.ordinal()];
121                    }
122            };
123    
124            @Override
125            public Collection<V> values() {
126                    final List<V> list = new ArrayList<V>(size());
127                    enumSet.runLoop(taskGet, list);
128                    return list;
129            }
130    
131            @Override
132            public Set<K> keySet() {
133                    return enumSet;
134            }
135    
136            @Override
137            public boolean containsKey(final Object key) {
138                    return enumSet.contains(key);
139            }
140    
141            // #####################################################################
142    
143            @Override
144            public boolean containsValue(final Object value) {
145                    throw new UnsupportedOperationException();
146            }
147    
148            @Override
149            public Set<Entry<K, V>> entrySet() {
150                    throw new UnsupportedOperationException();
151            }
152    
153            @Override
154            public void putAll(final Map<? extends K, ? extends V> m) {
155                    throw new UnsupportedOperationException();
156            }
157    
158            // #####################################################################
159    
160    }
