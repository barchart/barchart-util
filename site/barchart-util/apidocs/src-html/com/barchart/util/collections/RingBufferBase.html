008    package com.barchart.util.collections;
009    
010    import java.lang.reflect.Array;
011    import java.util.HashMap;
012    import java.util.LinkedList;
013    import java.util.List;
014    import java.util.Map;
015    
016    import com.barchart.util.anno.NotThreadSafe;
017    
018    /**
019     * http://en.wikipedia.org/wiki/Circular_buffer
020     * 
021     * legend:
022     * 
023     * absolute :: from beginning of array
024     * 
025     * clue :: absolute offset of a current item
026     * 
027     * mark :: absolute offset of a head (first) item
028     * 
029     * diff :: relative to mark offset of current item
030     * 
031     * head :: external index of a first logical item
032     * 
033     * tail :: external index of the last logical item
034     * 
035     * index :: external absolute index of current item
036     * 
037     * offset :: relative to head index of current item
038     * 
039     * place :: logical numbering of non-empty items, starts from 1
040     * 
041     * rules:
042     * 
043     * diff == clue - mark;
044     * 
045     * offset = index - head;
046     * 
047     * clue == (offset + mark) % length;
048     * 
049     * index == head + (diff >= 0 ? diff : diff + length );
050     * 
051     **/
052    @NotThreadSafe
053    public abstract class RingBufferBase<V> implements RingBuffer<V> {
054    
055            //
056    
057            protected abstract V arrayGet(int clue);
058    
059            protected abstract void arraySet(int clue, V value);
060    
061            protected abstract boolean isEmpty(int clue);
062    
063            @Override
064            public abstract int length();
065    
066            //
067    
068            /** internal index of the first item */
069            private int mark;
070    
071            /** external index of the first item */
072            private int head;
073    
074            //
075    
076            protected final int mark() {
077                    return mark;
078            }
079    
080            @Override
081            public final int head() {
082                    return head;
083            }
084    
085            @Override
086            public final int tail() {
087                    return head + length() - 1;
088            }
089    
090            protected final boolean isValidRange(final int value) {
091                    return 0 <= value && value < length();
092            }
093    
094            /** logical offset of clue from the mark */
095            protected final int offsetFromClue(final int clue) {
096                    assert isValidRange(clue) : " clue=" + clue;
097                    final int diff = clue - mark;
098                    return diff >= 0 ? diff : diff + length();
099            }
100    
101            /** external index for a clue */
102            protected final int indexFromClue(final int clue) {
103                    return head + offsetFromClue(clue);
104            }
105    
106            /** external index for a logical offset */
107            protected final int indexFromOffset(final int offset) {
108                    assert isValidRange(offset) : " offset=" + offset;
109                    return head + offset;
110            }
111    
112            /** absolute clue from logical offset off mark */
113            protected final int clueFromOffset(final int offset) {
114                    final int size = length();
115                    final int mark = this.mark;
116                    assert isValidRange(mark) : " mark=" + mark;
117                    assert isValidRange(offset) : " offset=" + offset;
118                    int clue = mark + offset;
119                    if (clue >= size) {
120                            clue -= size;
121                    }
122                    assert isValidRange(clue) : " clue=" + clue;
123                    return clue;
124            }
125    
126            /** absolute clue from absolute index */
127            protected final int clueFromIndex(final int index) {
128                    return clueFromOffset(index - head());
129            }
130    
131            @Override
132            public final boolean isValidIndex(final int index) {
133                    final int head = this.head;
134                    final int tail = head + length() - 1;
135                    return head <= index && index <= tail;
136            }
137    
138            //
139    
140            @Override
141            public final V get(final int index) {
142    
143                    final int offset = index - head;
144                    final int size = length();
145    
146                    if (offset < 0 || size <= offset) {
147                            throw new ArrayIndexOutOfBoundsException(" index=" + index
148                                            + " head=" + head() + " tail=" + tail());
149                    }
150    
151                    int clue = mark + offset;
152                    if (clue >= size) {
153                            clue -= size;
154                    }
155    
156                    assert isValidRange(clue) : " clue=" + clue;
157    
158                    return arrayGet(clue);
159    
160            }
161    
162            @Override
163            public final void set(final int index, final V value) {
164    
165                    final int offset = index - head;
166                    final int size = length();
167    
168                    if (offset < 0 || size <= offset) {
169                            throw new ArrayIndexOutOfBoundsException(" index=" + index
170                                            + " head=" + head() + " tail=" + tail());
171                    }
172    
173                    int clue = mark + offset;
174                    if (clue >= size) {
175                            clue -= size;
176                    }
177    
178                    assert isValidRange(clue) : " clue=" + clue;
179    
180                    arraySet(clue, value);
181    
182            }
183    
184            @Override
185            public final void setHead(final int index, final V value) {
186                    if (index == head()) {
187                            set(index, value);
188                    } else {
189                            setMark(0, index, value);
190                    }
191            }
192    
193            @Override
194            public final void setTail(final int index, final V value) {
195                    if (index == tail()) {
196                            set(index, value);
197                    } else {
198                            setMark(length() - 1, index, value);
199                    }
200            }
201    
202            protected final void setMark(final int zero, final int index, final V value) {
203    
204                    assert isValidRange(zero) : " zero=" + zero;
205    
206                    final int size = length();
207    
208                    int head = this.head;
209    
210                    int offset = index - (head + zero);
211    
212                    if (size <= Math.abs(offset)) {
213                            clear(index - zero);
214                            arraySet(zero, value);
215                            return;
216                    }
217    
218                    head += offset;
219    
220                    this.head = head;
221    
222                    int mark = this.mark;
223    
224                    assert isValidRange(mark) : " mark=" + mark;
225    
226                    while (offset < 0) {
227                            mark--;
228                            if (mark < 0) {
229                                    mark = size - 1;
230                            }
231                            arraySet(mark, null);
232                            offset++;
233                    }
234    
235                    assert isValidRange(mark) : " mark=" + mark;
236    
237                    while (offset > 0) {
238                            arraySet(mark, null);
239                            mark++;
240                            if (mark == size) {
241                                    mark = 0;
242                            }
243                            offset--;
244                    }
245    
246                    assert isValidRange(mark) : " mark=" + mark;
247    
248                    int clue = mark + zero;
249                    if (clue >= size) {
250                            clue -= size;
251                    }
252    
253                    assert isValidRange(clue) : " clue=" + clue;
254    
255                    arraySet(clue, value);
256    
257                    this.mark = mark;
258    
259            }
260    
261            @Override
262            public final void clear(final int head) {
263    
264                    final int size = length();
265    
266                    for (int clue = 0; clue < size; clue++) {
267                            arraySet(clue, null);
268                    }
269    
270                    this.mark = 0;
271                    this.head = head;
272    
273            }
274    
275            @Override
276            public final int count() {
277    
278                    final int size = length();
279    
280                    int count = 0;
281    
282                    for (int clue = 0; clue < size; clue++) {
283    
284                            if (isEmpty(clue)) {
285                                    continue;
286                            }
287    
288                            count++;
289    
290                    }
291    
292                    return count;
293    
294            }
295    
296            @SuppressWarnings("unchecked")
297            @Override
298            public final V[] asArray(final Class<V> klaz) {
299    
300                    final int size = length();
301                    final int mark = mark();
302    
303                    final V[] clone = (V[]) Array.newInstance(klaz, size);
304    
305                    for (int spot = 0; spot < size; spot++) {
306    
307                            int clue = mark + spot;
308                            if (clue >= size) {
309                                    clue -= size;
310                            }
311    
312                            assert isValidRange(clue) : " clue=" + clue;
313    
314                            if (isEmpty(clue)) {
315                                    continue;
316                            }
317    
318                            clone[spot] = arrayGet(clue);
319    
320                    }
321    
322                    return clone;
323    
324            }
325    
326            @Override
327            public final List<V> asList() {
328    
329                    final int size = length();
330                    final int mark = mark();
331    
332                    final List<V> list = new LinkedList<V>();
333    
334                    for (int spot = 0; spot < size; spot++) {
335    
336                            int clue = mark + spot;
337                            if (clue >= size) {
338                                    clue -= size;
339                            }
340    
341                            assert isValidRange(clue) : " clue=" + clue;
342    
343                            if (isEmpty(clue)) {
344                                    continue;
345                            }
346    
347                            list.add(arrayGet(clue));
348    
349                    }
350    
351                    return list;
352    
353            }
354    
355            @Override
356            public final Map<Integer, V> asMap() {
357    
358                    final int size = length();
359                    final int head = head();
360                    final int mark = mark();
361    
362                    final Map<Integer, V> map = new HashMap<Integer, V>();
363    
364                    for (int spot = 0; spot < size; spot++) {
365    
366                            int clue = mark + spot;
367                            if (clue >= size) {
368                                    clue -= size;
369                            }
370    
371                            assert isValidRange(clue) : " clue=" + clue;
372    
373                            if (isEmpty(clue)) {
374                                    continue;
375                            }
376    
377                            map.put(head + spot, arrayGet(clue));
378    
379                    }
380    
381                    return map;
382    
383            }
384
