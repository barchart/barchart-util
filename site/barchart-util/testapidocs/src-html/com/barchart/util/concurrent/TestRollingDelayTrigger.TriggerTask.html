008    package com.barchart.util.concurrent;
009    
010    import static org.junit.Assert.assertEquals;
011    
012    import org.junit.Before;
013    import org.junit.Test;
014    
015    import com.barchart.util.test.CallableTest;
016    
017    public class TestRollingDelayTrigger {
018    
019            private TriggerTask task;
020            private RollingDelayTrigger trigger;
021    
022            @Before
023            public void setUp() {
024                    task = new TriggerTask();
025                    trigger = new RollingDelayTrigger(task, 100);
026            }
027    
028            @Test
029            public void testDelay() throws Exception {
030                    trigger.trigger();
031                    assertEquals(0, task.runCount);
032                    CallableTest.waitFor(new CallableTest.FieldValue(task, "runCount", 1));
033                    assertEquals(1, task.runCount);
034            }
035    
036            @Test
037            public void testMultipleTriggers() throws Exception {
038    
039                    for (int i = 0; i < 10; i++) {
040                            trigger.trigger();
041                            Thread.sleep(20);
042                    }
043    
044                    assertEquals(0, task.runCount);
045    
046                    CallableTest.waitFor(new CallableTest.FieldValue(task, "runCount", 1));
047                    assertEquals(1, task.runCount);
048    
049            }
050    
051            @Test
052            public void testMultipleTasks() throws Exception {
053    
054                    trigger.trigger();
055                    CallableTest.waitFor(new CallableTest.FieldValue(task, "runCount", 1));
056                    assertEquals(1, task.runCount);
057    
058                    trigger.trigger();
059                    CallableTest.waitFor(new CallableTest.FieldValue(task, "runCount", 2));
060                    assertEquals(2, task.runCount);
061    
062            }
063    
064            public static class TriggerTask implements Runnable {
065    
066                    public volatile int runCount = 0;
067    
068                    @Override
069                    public void run() {
070                            runCount++;
071                    }
072    
073            }
074    
075    }
