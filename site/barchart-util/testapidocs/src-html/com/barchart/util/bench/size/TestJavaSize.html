031            @Test
032            public void testX() {
033    
034                    assertEquals(of(null), 0);
035    
036                    assertEquals(of(true), 1);
037    
038                    assertEquals(of((byte) 37), 1);
039    
040                    assertEquals(of('x'), 2);
041    
042                    assertEquals(of((short) 37), 2);
043    
044                    assertEquals(of(37), 4);
045    
046                    assertEquals(of(37L), 8);
047    
048                    assertEquals(of(37.0f), 4);
049    
050                    assertEquals(of(37.0), 8);
051    
052                    assertEquals(of(new boolean[0]), OBJECT_SIZE);
053    
054                    assertEquals(of(new byte[10]), OBJECT_SIZE + 10);
055    
056                    assertEquals(of(new char[10]), OBJECT_SIZE + 10 * 2);
057    
058                    assertEquals(of(new char[8][9]), OBJECT_SIZE + 8
059                                    * (REFERENCE_SIZE + OBJECT_SIZE + 9 * 2));
060    
061                    // assertEquals(of(new double[5][6][7]), OBJECT_SIZE + 5*(REFERENCE_SIZE
062                    // + OBJECT_SIZE + 6));
063    
064                    assertEquals(of(new int[0][10]), OBJECT_SIZE);
065    
066                    assertEquals(of(new Object()), OBJECT_SIZE);
067    
068                    assertEquals(of(new String()), STRING_BASE);
069    
070                    assertEquals(of(new String[100]), OBJECT_SIZE + 100 * REFERENCE_SIZE);
071    
072                    assertEquals(of(new String[7][8]), OBJECT_SIZE + 7
073                                    * (REFERENCE_SIZE + OBJECT_SIZE + 8 * REFERENCE_SIZE));
074    
075                    final Object ov[] = new Object[3];
076                    ov[0] = new byte[10];
077                    ov[1] = null;
078                    ov[2] = new double[10];
079                    assertEquals(of(ov), 90 + 3 * REFERENCE_SIZE + 3 * OBJECT_SIZE);
080    
081                    final String sv[] = new String[10];
082                    for (int i = 0; i < 10; i++) {
083                            sv[i] = new String();
084                    }
085                    assertEquals(of(sv), OBJECT_SIZE + 10 * REFERENCE_SIZE + 10
086                                    * STRING_BASE);
087    
088                    assertEquals(of(new Integer(37)), 4 + OBJECT_SIZE + 4);
089    
090                    assertEquals(of(new test_Class1()), test_Class1.SIZE);
091    
092                    assertEquals(of(new test_Class2()), test_Class2.SIZE);
093    
094                    assertEquals(of(new test_Class3()), test_Class3.SIZE);

178            @Test
179            public void test5() {
180    
181                    final int row = 3;
182                    final int col = 7;
183    
184                    final int sizeL0 = of('A');
185                    final int sizeL1 = OBJECT_SIZE + col * sizeL0;
186                    final int sizeL2 = OBJECT_SIZE + row * (REFERENCE_SIZE + sizeL1);
187    
188                    final char[][] L2 = new char[row][col];
189                    assertEquals(of(L2), sizeL2);
190    
191                    for (final char[] L1 : L2) {
192                            assertNotNull(L1);
193                            assertEquals(of(L1), sizeL1);
194                            for (final char L0 : L1) {
195                                    assertEquals(0, L0);
196                                    assertEquals(of(L0), 2);
197                            }
198                    }
