194            @Test
195            public void testAsArray() {
196    
197                    final RingBuffer<Integer> ring = new RingBufferSimple<Integer>(10);
198                    assertEquals(ring.length(), 10);
199    
200                    ring.setHead(100, 100);
201    
202                    for (int k = 100; k < 110; k++) {
203                            ring.set(k, k);
204                    }
205    
206                    ring.set(102, null);
207                    ring.set(107, null);
208                    assertEquals(ring.count(), 8);
209    
210                    final Integer[] clone1 = ring.asArray(Integer.class);
211                    final Integer[] clone2 = ring.asArray(Integer.class);
212    
213                    assertFalse(clone1 == clone2);
214                    assertTrue(Arrays.equals(clone1, clone2));
215    
216                    for (int k = 0; k < 10; k++) {
217                            assertEquals(ring.get(100 + k), clone1[k]);
218                            assertEquals(ring.get(100 + k), clone2[k]);
219                    }
220    
221            }

223            @Test
224            public void testAsList() {
225    
226                    final RingBuffer<Integer> ring = new RingBufferSimple<Integer>(10);
227                    assertEquals(ring.length(), 10);
228    
229                    ring.setHead(100, 100);
230    
231                    for (int k = 100; k < 110; k++) {
232                            ring.set(k, k);
233                    }
234    
235                    ring.set(102, null);
236                    ring.set(107, null);
237                    assertEquals(ring.count(), 8);
238    
239                    final List<Integer> list = ring.asList();
240    
241                    assertEquals(list.size(), ring.count());
242    
243                    int index = 0;
244                    for (int k = 100; k < 110; k++) {
245                            final Integer ringItem = ring.get(k);
246                            if (ringItem == null) {
247                                    continue;
248                            }
249                            final Integer listItem = list.get(index++);
250                            assertEquals(ringItem, listItem);
251                    }
252    
253            }
