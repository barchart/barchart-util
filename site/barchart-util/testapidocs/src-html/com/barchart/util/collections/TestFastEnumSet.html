008    package com.barchart.util.collections;
009    
010    import static java.lang.System.*;
011    import static org.junit.Assert.*;
012    
013    import java.util.Set;
014    
015    import org.junit.After;
016    import org.junit.Before;
017    import org.junit.Test;
018    
019    public class TestFastEnumSet {
020    
021            @Before
022            public void setUp() throws Exception {
023            }
024    
025            @After
026            public void tearDown() throws Exception {
027            }
028    
029            enum E1 implements BitSetEnum<E1> {
030    
031                    a, b, c;
032    
033                    private final long mask = ONE << ordinal();
034    
035                    @Override
036                    public long mask() {
037                            return mask;
038                    }
039    
040                    private final static E1[] VALUES = values();
041    
042                    public static E1[] valuesUnsafe() {
043                            return VALUES;
044                    }
045    
046            }
047    
048            @Test
049            public void testBitSetFast0() {
050    
051                    // empty
052                    final Set<E1> set = new FastEnumSet<E1>(E1.values());
053    
054                    int count;
055    
056                    count = 0;
057                    for (final E1 event : set) {
058                            count++;
059                            out.println(event);
060                    }
061                    assertEquals(count, 0);
062    
063                    set.add(E1.a);
064                    set.add(E1.c);
065    
066                    count = 0;
067                    for (final E1 event : set) {
068                            count++;
069                            out.println(event.mask());
070                    }
071                    assertEquals(count, 2);
072    
073            }
074    
075            @Test
076            public void testBitSetFast1() {
077    
078                    final FastEnumSet<E1> eventSet1 = new FastEnumSet<E1>(E1.valuesUnsafe());
079                    final FastEnumSet<E1> eventSet2 = new FastEnumSet<E1>(E1.valuesUnsafe());
080    
081                    assertTrue(eventSet1.enumValues == eventSet2.enumValues);
082    
083            }
084    }
