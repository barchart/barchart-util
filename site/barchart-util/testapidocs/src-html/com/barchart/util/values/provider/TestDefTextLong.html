029            @Test
030            public void testLongConstructor() {
031    
032                    final long value = 65 << 0 + 66 << 8;
033    
034                    final TextValue text = new DefTextLong(value);
035    
036                    final int size = JavaSize.of(text);
037    
038                    // System.err.println("size=" + size);
039    
040                    assertEquals(size, 16);
041            }

043            @Test
044            public void testCharAt() {
045    
046                    final long value = (0x41 << 0 * 8) + (0x42 << 1 * 8) + (0x43 << 2 * 8);
047    
048                    // System.err.println("value=" + Long.toHexString(value));
049    
050                    final TextValue text = new DefTextLong(value);
051    
052                    assertEquals(text.length(), 3);
053    
054                    assertEquals(text.charAt(0), 'A');
055                    assertEquals(text.charAt(1), 'B');
056                    assertEquals(text.charAt(2), 'C');
057    
058                    // System.err.println("size=" + size);
059    
060            }

062            @Test
063            public void testHashCode() {
064    
065                    final long value = (0x41 << 0 * 8) + (0x42 << 1 * 8) + (0x43 << 2 * 8);
066    
067                    // System.err.println("value=" + Long.toHexString(value));
068    
069                    final TextValue text = new DefTextLong(value);
070    
071                    assertEquals(text.hashCode(), "ABC".hashCode());
072    
073                    assertTrue(text.equals("ABC"));
074    
075            }
076    
077            @Test
078            public void testSubString() {
079    
080                    final long value = (0x41 << 0 * 8) + (0x42 << 1 * 8) + (0x43 << 2 * 8)
081                                    + (0x44 << 3 * 8);
082    
083                    // System.err.println("value=" + Long.toHexString(value));
084    
085                    //
086    
087                    final TextValue text0 = new DefTextLong(value);
088                    // System.err.println("text0=" + text0);
089                    assertEquals(text0.hashCode(), "ABCD".hashCode());
090    
091                    //
092    
093                    final CharSequence seq1 = text0.subSequence(0, 1);
094                    // System.err.println("text1=" + seq1);
095                    assertTrue(seq1.equals("A"));
096    
097                    //
098    
099                    final CharSequence seq2 = text0.subSequence(1, 3);
100                    // System.err.println("text2=" + seq2);
101                    assertTrue(seq2.equals("BC"));
102                    assertEquals(seq2.length(), 2);
103    
104                    //
105    
106                    final CharSequence seq3 = text0.subSequence(3, 4);
107                    // System.err.println("text3=" + seq3);
108                    assertTrue(seq3.equals("D"));
109    
110            }
111

132            public void testStringConstructor() {
133    
134                    final String string = "ABC";
135    
136                    // System.err.println("value=" + Long.toHexString(value));
137    
138                    final TextValue text = new DefTextLong(string);

155            @Test
156            public void testToUpperCase() {
157    
158                    final TextValue text1 = new DefTextLong("aBcD1?xY");
159                    final TextValue text2 = text1.toUpperCase();
160                    // System.err.println("text1=" + text1);
161                    // System.err.println("text2=" + text2);
162    
163                    assertEquals(new DefTextLong("").toUpperCase(), "");
164                    assertEquals(new DefTextLong("a").toUpperCase(), "A");
165                    assertEquals(new DefTextLong("aBcD1?xY").toUpperCase(), "ABCD1?XY");
166    
167            }
168    
169            @Test
170            public void testToLowerCase() {
171    
172                    assertEquals(new DefTextLong("").toLowerCase(), "");
173    
174                    assertEquals(new DefTextLong("a").toLowerCase(), "a");
175                    assertEquals(new DefTextLong("A").toLowerCase(), "a");
176    
177                    assertEquals(new DefTextLong("aBcD1?xY").toLowerCase(), "abcd1?xy");
178    
179            }
