039                    assertEquals(p1.compareTo(p2), 0);
040                    assertEquals(p2.compareTo(p1), 0);
041    
042                    assertEquals(JavaSize.of(ValueConst.NULL_PRICE), 8);
043                    assertEquals(JavaSize.of(p0), 16);
044                    assertEquals(JavaSize.of(p1), 16);
045                    assertEquals(JavaSize.of(p2), 24);
046    
047                    assertTrue(newPrice(0, 0) instanceof NulPrice);
048    
049                    assertTrue(newPrice(12, 0) instanceof DefPrice0);

051                    assertTrue(newPrice(121212121212L, 0) instanceof DefPrice0);
052    
053                    assertTrue(newPrice(12, 1) instanceof DefPriceA);
054                    assertTrue(newPrice(122342341, -123) instanceof DefPriceA);
055    
056                    assertTrue(newPrice(121212121212L, -112) instanceof DefPriceB);

058            }
