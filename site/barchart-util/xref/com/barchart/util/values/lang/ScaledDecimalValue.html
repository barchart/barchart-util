25  	@SuppressWarnings("unchecked")
26  	@Override
27  	public final T norm() {
28  
29  		long m = mantissa();
30  		int e = exponent();
31  
32  		final int p = e;
33  
34  		while (m % 10 == 0) {
35  			m /= 10;
36  			e++;
37  		}
38  
39  		if (p == e) {
40  			return (T) this;
41  		} else {
42  			return result(m, e);
43  		}
44  
45  	}

49  	@Override
50  	public final T scale(final int exponent) throws ArithmeticException {
51  
52  		long m = mantissa();
53  		int e = exponent();
54  
55  		if (e == exponent) {
56  			return (T) this;
57  		}
58  
59  		while (e > exponent) {
60  			m = MathExtra.longMult10(m);
61  			e--;
62  		}
63  		while (e < exponent) {
64  			m /= 10;
65  			e++;
66  		}
67  
68  		return result(m, e);
69  
70  	}

72  	@Override
73  	public final int compareTo(final T that) {
74  
75  		int e1 = this.exponent();
76  		int e2 = that.exponent();
77  
78  		long m1 = this.mantissa();
79  		long m2 = that.mantissa();
80  
81  		if (e1 != e2) {
82  			if ((m1 ^ m2) < 0) {
83  				// different sign
84  				return m1 >= 0 ? +1 : -1;
85  			} else {
86  				// identical sign
87  				final int e = max(e1, e2);
88  				while (e > e1) {
89  					m1 /= 10L;
90  					e1++;
91  				}
92  				while (e > e2) {
93  					m2 /= 10L;
94  					e2++;
95  				}
96  			}
97  
98  		}
99  
100 		return m1 < m2 ? -1 : (m1 == m2 ? 0 : +1);
