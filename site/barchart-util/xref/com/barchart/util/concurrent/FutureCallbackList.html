13  /**
14   * Groups a number of {@link FutureCallbackTask} results together, and executes
15   * a single callback only after all FutureCallbacks have completed (succeeded or
16   * failed.)
17   * 
18   * This class does not support direct access to succeed() or fail(), and will
19   * throw and UnsupportedOperationException for them.
20   * 
21   * @author jeremy
22   * @see FutureCallbackTask
23   * @param <E>
24   *            The deferred result type
25   */
26  public class FutureCallbackList<E> implements
27  		FutureCallback<List<E>, FutureCallbackList<E>> {
28  
29  	private final static Logger log = LoggerFactory
30  			.getLogger(FutureCallbackList.class);
31  
32  	private final FutureListener<E> listener = new FutureListener<E>() {
33  
34  		@Override
35  		public void resultAvailable(final Future<E> result_) {
36  			onResponse(result_);
37  		}
38  
39  	};
40  
41  	private final List<FutureListener<List<E>>> listeners =
42  			new CopyOnWriteArrayList<FutureListener<List<E>>>();
43  	private final List<E> responses = new CopyOnWriteArrayList<E>();
44  	private final List<? extends FutureCallback<E, ?>> results;
45  	private final Semaphore resultMonitor = new Semaphore(1);
46  
47  	private volatile int completed = 0;
48  	private volatile boolean fired = false;
49  	private volatile boolean cancelled = false;
50  
51  	/**
52  	 * Create a FutureCallbackList from the specified FutureCallback objects.
53  	 * 
54  	 * @param deferreds_
55  	 *            The list of deferred results to monitor
56  	 */
57  	public FutureCallbackList(
58  			final List<? extends FutureCallback<E, ?>> results_) {
59  		results = results_;
60  		if (results_ == null || results_.size() == 0) {
61  			fired = true;
62  		} else {
63  			// Lock the semaphore until a result is available
64  			resultMonitor.acquireUninterruptibly();
65  			for (final FutureCallback<E, ?> r : results_) {
66  				r.addResultListener(listener);
67  			}
68  		}
69  	}
70  
71  	/**
72  	 * Add a callback listener that will be called when all FutureCallbacks have
73  	 * completed. The callback will be passed a list of results collected from
74  	 * the FutureCallbacks. A null member may indicate that a FutureCallback
75  	 * failed, but does not guarantee it (since a FutureCallback callback could
76  	 * return a null value on purpose.)
77  	 * 
78  	 * @param callback
79  	 *            The object to notify of the deferred results
80  	 */
81  	@Override
82  	public FutureCallbackList<E> addResultListener(
83  			final FutureListener<List<E>> callback) {
84  		listeners.add(callback);
85  		if (fired) {
86  			try {
87  				callback.resultAvailable(this);
88  			} catch (final Exception e) {
89  				log.warn("Unhandled exception in callback", e);
90  			}
91  		}
92  		return this;
93  	}
94  
95  	/**
96  	 * Register a response from a FutureCallback.
97  	 */
98  	private void onResponse(final Future<E> result) {
99  		if (fired) {
100 			return;
101 		}
102 		completed++;
103 		if (completed == results.size()) {
104 			for (final FutureCallback<E, ?> r : results) {
105 				try {
106 					responses.add(r.get());
107 				} catch (final Exception e) {
108 					responses.add(null);
109 				}
110 			}
111 			fired = true;
112 			resultMonitor.release();
113 			for (final FutureListener<List<E>> l : listeners) {
114 				try {
115 					l.resultAvailable(this);
116 				} catch (final Exception e) {
117 					log.warn("Unhandled exception in callback", e);
118 				}
119 			}
120 		}
121 	}
122 
123 	@Override
124 	public boolean cancel(final boolean interrupt) {
125 		if (fired) {
126 			return false;
127 		}
128 		cancelled = true;
129 		for (final FutureCallback<E, ?> r : results) {
130 			r.cancel(interrupt);
131 		}
132 		return true;
133 	}
134 
135 	@Override
136 	public List<E> get() {
137 		if (!fired) {
138 			resultMonitor.acquireUninterruptibly();
139 		}
140 		return responses;
141 	}
142 
143 	@Override
144 	public List<E> getUnchecked() {
145 		return get();
146 	}
147 
148 	@Override
149 	public List<E> get(final long timeout, final TimeUnit unit)
150 			throws InterruptedException, TimeoutException {
151 		if (!fired && !resultMonitor.tryAcquire(timeout, unit)) {
152 			throw new TimeoutException("Timed out waiting for result");
153 		}
154 		return responses;
155 	}
156 
157 	@Override
158 	public boolean isCancelled() {
159 		return cancelled;
160 	}

164 		return fired;
