8   package com.barchart.util.collections;

10  import com.barchart.util.math.MathExtra;
11  import com.barchart.util.values.lang.ScaledDecimal;
12  
13  public abstract class ScadecRingBufferBase<K extends ScaledDecimal<K, ?>, V>
14  		extends RingBufferBase<V> implements ScadecRingBuffer<K, V> {
15  
16  	@Override
17  	public abstract K keyStep();
18  
19  	//
20  
21  	protected final void checkKeyStep(final K keyStep) {
22  		if (keyStep == null || keyStep.isZero()) {
23  			throw new IllegalArgumentException(
24  					"keyStep == null || keyStep.isZero()");
25  		}
26  	}
27  
28  	@Override
29  	public final K keyHead() {
30  		return keyStep().mult(head());
31  	}
32  
33  	@Override
34  	public final K keyTail() {
35  		return keyStep().mult(tail());
36  	}
37  
38  	//
39  
40  	@Override
41  	public final int index(final K key) {
42  		final long count = key.count(keyStep());
43  		final int index = MathExtra.castLongToInt(count);
44  		return index;
45  	}
46  
47  	@Override
48  	public final V get(final K key) {
49  		return get(index(key));
50  	}
51  
52  	@Override
53  	public final void set(final K key, final V value) {
54  		set(index(key), value);
55  	}
56  
57  	@Override
58  	public final void setHead(final K key, final V value) {
59  		setHead(index(key), value);
60  	}
61  
62  	@Override
63  	public final void setTail(final K key, final V value) {
64  		setTail(index(key), value);
65  	}
66  
67  }
